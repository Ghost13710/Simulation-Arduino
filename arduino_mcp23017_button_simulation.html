// MCP23017 + Arduino — bouton → Serial
#include <Wire.h>
#include <Adafruit_MCP23X17.h>

Adafruit_MCP23X17 mcp;
const uint8_t BTN_PIN = 0;  // GPA0
bool lastState = HIGH;

void setup() {
  Serial.begin(9600);
  mcp.begin_I2C();
  mcp.pinMode(BTN_PIN, INPUT_PULLUP);
}

void loop() {
  bool state = mcp.digitalRead(BTN_PIN);
  if (state != lastState) {
    Serial.println(state == LOW ? 1 : 0);
    lastState = state;
  }
  delay(10);
}
